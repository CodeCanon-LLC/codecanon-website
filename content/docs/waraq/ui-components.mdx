All components are imported from `@codecanon/waraq/ui` and can be used independently of the editor — drop them into any Tailwind + ShadCN project.

```ts
import { Button, Input, ColorPicker, /* … */ } from "@codecanon/waraq/ui"
```

---

## Button

```tsx
import { Button } from "@codecanon/waraq/ui"

<Button>Default</Button>
<Button variant="secondary">Secondary</Button>
<Button variant="outline">Outline</Button>
<Button variant="ghost">Ghost</Button>
<Button variant="destructive">Destructive</Button>
<Button variant="link">Link</Button>
```

### Sizes

```tsx
<Button size="sm">Small</Button>
<Button size="default">Default</Button>
<Button size="lg">Large</Button>
<Button size="icon"><Pencil size={16} /></Button>
```

| Prop | Type | Default |
| --- | --- | --- |
| `variant` | `"default" \| "secondary" \| "outline" \| "ghost" \| "destructive" \| "link"` | `"default"` |
| `size` | `"default" \| "sm" \| "lg" \| "icon" \| "icon-sm" \| "icon-lg"` | `"default"` |
| `asChild` | `boolean` | `false` |
| `tooltip` | `string` | — |
| `tooltipSide` | `"top" \| "right" \| "bottom" \| "left"` | — |
| `tooltipShortcut` | `string[]` | — |

## InputNumber

Number input with optional stepper controls, min/max, suffix/prefix, and decimal scale.

```tsx
import { InputNumber } from "@codecanon/waraq/ui"

<InputNumber
  defaultValue={42}
  suffix="px"
  showControls
  min={0}
  max={1000}
/>
```

| Prop | Type | Description |
| --- | --- | --- |
| `value` | `number \| undefined` | Controlled value |
| `onValueChange` | `(value: number \| undefined) => void` | Change handler |
| `min` / `max` | `number` | Clamp bounds |
| `stepper` | `number` | Step increment |
| `suffix` / `prefix` | `string` | Unit label |
| `showControls` | `boolean` | Show +/– spinner buttons |
| `decimalScale` | `number` | Number of decimal places |

## ColorPicker

Full color picker with hex input and optional eye dropper.

```tsx
import { ColorPicker } from "@codecanon/waraq/ui"

<ColorPicker
  value="#3b82f6"
  onChange={(color) => console.log(color)}
  showInput
  showPreview
/>
```

| Prop | Type | Default | Description |
| --- | --- | --- | --- |
| `value` | `string` | — | Controlled hex color |
| `onChange` | `(color: string) => void` | — | Change handler |
| `showInput` | `boolean` | `false` | Show hex text input |
| `showPreview` | `boolean` | `false` | Show color swatch preview |
| `disabled` | `boolean` | `false` | Disable interaction |

### Compact variants

```tsx
import { ColorPickerCompact, ColorPreview, ColorInput } from "@codecanon/waraq/ui"

// Compact trigger button — opens picker in a popover
<ColorPickerCompact value="#f59e0b" onChange={setColor} />

// Color swatch only
<ColorPreview value="#f59e0b" />

// Hex text input only
<ColorInput value="#f59e0b" onChange={setColor} />
```

## Dialog

```tsx
import {
  Dialog, DialogTrigger, DialogContent,
  DialogHeader, DialogTitle, DialogDescription, DialogFooter,
} from "@codecanon/waraq/ui"
import { Button } from "@codecanon/waraq/ui"

<Dialog>
  <DialogTrigger asChild>
    <Button variant="outline">Open dialog</Button>
  </DialogTrigger>
  <DialogContent>
    <DialogHeader>
      <DialogTitle>Are you sure?</DialogTitle>
      <DialogDescription>This action cannot be undone.</DialogDescription>
    </DialogHeader>
    <DialogFooter>
      <Button variant="outline">Cancel</Button>
      <Button variant="destructive">Confirm</Button>
    </DialogFooter>
  </DialogContent>
</Dialog>
```
